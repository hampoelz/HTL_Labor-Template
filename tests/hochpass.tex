\subsection{Berechnungen}

\begin{sagesilent}
    L = R/(2*pi*f_g)
\end{sagesilent}

\begin{align*}
    f_{g} &= \frac{R}{1 \pi L} \\
    \Rightarrow L &= \frac{R}{2 \pi f_{g}} \\
    L &= \frac{\var{R}}{2 \pi \var{f_g}} \\
    L &= \var{L, 3} H
\end{align*}
